che restituisca la lista degli archi in avanti del grafo

time = 0 
dfs(G, k){
    
    list = new list()
	for nodes in G.nodes{
		node.color = white
		node.d = 0
		node.f = 0
		node.parent = null
	}
	
	for node in G.nodes{
		if node.color == white{
			dfs_visit(node, list)
		}
	}

    return list
}
